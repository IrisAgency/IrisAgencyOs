// 3. Add string value: calendar.view
// 4. Save

===== ALTERNATIVE: Run this in your app's browser console while logged in as ADMIN =====

const db = firebase.firestore();

// Find and update Copywriter role
db.collection('roles').where('name', '==', 'Copywriter').get()
  .then(snapshot => {
    if (snapshot.empty) {
      console.error('❌ Copywriter role not found');
      return;
    }
    
    const doc = snapshot.docs[0];
    const data = doc.data();
    const permissions = data.permissions || [];
    
    console.log('📋 Current permissions:', permissions);
    
    if (permissions.includes('calendar.view')) {
      console.log('✅ calendar.view already exists');
      return;
    }
    
    // Add calendar.view before calendar.manage
    const newPermissions = [...permissions];
    const manageIndex = newPermissions.indexOf('calendar.manage');
    if (manageIndex !== -1) {
      newPermissions.splice(manageIndex, 0, 'calendar.view');
    } else {
      newPermissions.push('calendar.view');
    }
    
    // Update Firestore
    return doc.ref.update({
      permissions: newPermissions,
      updatedAt: new Date().toISOString()
    }).then(() => {
      console.log('✅ Successfully updated Copywriter role');
      console.log('📋 New permissions:', newPermissions);
      console.log('🔄 Please refresh the page');
    });
  })
  .catch(error => {
    console.error('❌ Error:', error);
  });
